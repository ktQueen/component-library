使用方法：

        
html

            <div class="box box1">
                <div class="header header1">
            </div>
        
js调用

        require([
            'jquery',
            'modules/drag/js/drag'
        ], function($, drag) {
            //第一个参数是触发拖拽的元素，第二个参数是被拖拽的元素
            drag($('.header1'),$('.box1'));
        });
