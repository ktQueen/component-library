使用默认数据的方式

demo示例

代码示例

        html部分:
            <div id="demo1"></div>
            <script data-main="你自己的js文件路径" src="modules/libs/rq/require.js"></script>
        js部分:
            require.config({
                baseUrl: 'http://localhost:63342/github/component-library/'
            });
            require([
                'modules/city_selector/js/city_selector',
                'modules/city_selector/js/city_data'//默认数据文件
            ], function(CitySelector,cityData) {
                new CitySelector({
                    element:$('#demo1'),
                    data:cityData,//数据
                    isOwnData:true,//是否是默认数据
                    required:false,//是否是必填
                });
            });

使用自定义数据的方式

方法演示

demo示例

代码示例

        html部分:
            <div id="demo2"></div>
            <script data-main="你自己的js文件路径" src="modules/libs/rq/require.js"></script>
        js部分:
            require.config({
                baseUrl: 'http://localhost:63342/github/component-library/'
            });
            require([
                'modules/city_selector/js/city_selector'
            ], function(CitySelector) {
                new CitySelector({
                    element:$('#demo2'),
                    multiSelect:true,
                    idFlag:'pingyin',//标识字段
                    nameFlag:'cityName',//名字字段
                    pinyinFlag:'pingyin',//拼音字段
                    hotFlag:'isHot',//热门城市字段
                    data:function(callback){//获取自定义数据
                        $.ajax({
                            url:'js/city.json',
                            dataType:'json',
                            success:function(data){
                                callback(data.repData.citySubjectResultList);
                            },
                            error:function(){}
                        });
                    }
                });
            });
